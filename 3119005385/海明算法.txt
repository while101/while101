 public double getDistance(String str1, String str2) { 
       int distance; 
           distance = 0; 
           for (int i = 0; i < str1.length(); i++) { 
               if (str1.charAt(i) != str2.charAt(i)) { 
                   distance++; 
               }   
       } 
       return 0.01*(100-distance*100/1024); 
    }
